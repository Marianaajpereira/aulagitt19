-- Montar o select de relacionamento entre as tabelas utilizando o inner join
select f.matricula,f.nome_funcionario,c.nome_cargo,c.valor_cargo
from itau.cargo as  c

on(f.cod_cargo_funcionario=c.cod_cargo);
